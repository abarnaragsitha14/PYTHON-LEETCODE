Two sum IV input is a bst
class Solution:
    def findTarget(self, root: Optional[TreeNode], k: int) -> bool:
        res = set()

        def dfs(node):
            if not node:
                return False

            if (k - node.val) in res:
                return True

            res.add(node.val)

            return dfs(node.left) or dfs(node.right)

        return dfs(root)
