class Solution(object):
    def findMode(self, root):
        freq = {}

        def inorder(node):
            if not node:
                return
            inorder(node.left)
            freq[node.val] = freq.get(node.val, 0) + 1
            inorder(node.right)

        inorder(root)

        max_freq = max(freq.values())
        return [k for k, v in freq.items() if v == max_freq]
